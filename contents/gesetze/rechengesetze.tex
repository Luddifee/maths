\section{Rechengesetze}
\subsection{Kommutativgesetz}
\[ a + b = b + a \]

\subsection{Assoziativgesetz}
\[ a + b + c = a + (b + c) = (a + b) + c \]

\subsection{Distributivgesetz}
\subsubsection{Ausmultiplizieren}
\[ a \cdot (b + c) = ab + bc \]

\subsubsection{Ausklammern}
\[ ab + ac = a \cdot (b + c)  \]
